@startuml
' Diagrama de Contexto - Bankify (PlantUML moderno)
title Diagrama de Contexto - Bankify

left to right direction
skinparam packageStyle rectangle

actor Cliente
actor "Asesor / Operador" as Asesor
actor Supervisor
actor "Gerente Financiero" as Gerente
actor DIAN
actor "Banco (registro y verificación)" as BancoExterno

package "Sistema Bankify" as Bankify {
  rectangle "Bankify\n(Sistema de gestión de cuentas)" as S
}

Cliente --> S : Consultar saldo\nRealizar depósito\nGenerar reporte (PDF)
Asesor --> S : Crear/Actualizar cuentas\nActivar/Inactivar cuentas
Supervisor --> S : Gestionar clientes
Gerente --> S : Solicitar reportes\nRevisar reportes
S --> DIAN : Envío de reportes (JSON)
S --> BancoExterno : Validar código de banco
BancoExterno --> S : Confirmación / Rechazo de validación

@enduml
